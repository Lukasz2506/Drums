for(var i=0;i<document.querySelectorAll(".drum").length;i++){var singleDrum=document.querySelectorAll(".drum")[i];singleDrum.addEventListener("click",(function(){var e=this.innerHTML;makeSound(e),console.log(this),buttonAnimation(e)}))}function makeSound(e){switch(e){case"w":new Audio("./sounds/tom-1.mp3").play();break;case"a":new Audio("./sounds/tom-2.mp3").play();break;case"s":new Audio("./sounds/tom-3.mp3").play();break;case"d":new Audio("./sounds/tom-4.mp3").play();break;case"j":new Audio("./sounds/snare.mp3").play();break;case"k":new Audio("./sounds/crash.mp3").play();break;case"l":new Audio("./sounds/kick-bass.mp3").play();break;default:console.log(e)}}function buttonAnimation(e){var n=document.querySelector("."+e);n.classList.add("pressed"),setTimeout((function(){n.classList.remove("pressed")}),100)}document.addEventListener("keypress",(function(e){var n=e.key.toLowerCase();makeSound(n),buttonAnimation(n)}));
